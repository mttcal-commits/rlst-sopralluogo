let step = 0; const totalSteps = 7; // Funzione principale per girare le pagine function goTo(n) { // Nasconde la sezione corrente const currentSec = document.getElementById('s' + step); if (currentSec) currentSec.classList.remove('act'); const currentPill = document.querySelectorAll('.sp')[step]; if (currentPill) currentPill.classList.remove('act'); // Aggiorna lo stato dello step step = n; // Mostra la nuova sezione const newSec = document.getElementById('s' + step); if (newSec) newSec.classList.add('act'); const pills = document.querySelectorAll('.sp'); if (pills[step]) { pills[step].classList.add('act'); pills[step].scrollIntoView({ behavior: 'smooth', inline: 'center' }); } // Aggiorna i pulsanti di navigazione inferiore const btnPrev = document.getElementById('btnP'); if (btnPrev) btnPrev.style.display = step === 0 ? 'none' : 'flex'; const btnNext = document.getElementById('btnN'); if (btnNext) { btnNext.textContent = step === 6 ? 'GENERA PDF' : 'AVANTI →'; btnNext.className = step === 6 ? 'btn btn-ok' : 'btn btn-p'; } // Aggiorna contatore e barra progresso document.getElementById('sc').textContent = (step + 1) + ' / ' + totalSteps; document.getElementById('pb').style.width = ((step + 1) / totalSteps * 100) + '%'; window.scrollTo(0, 0); } function nextStep() { if (step < 6) { const pills = document.querySelectorAll('.sp'); if (pills[step]) pills[step].classList.add('done'); goTo(step + 1); } else { alert("Funzione PDF in attivazione..."); } } function prevStep() { if (step > 0) goTo(step - 1); }
